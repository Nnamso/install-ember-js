# install-ember-js
How to setup development environment of ember JS on Mac OS Catalina

install node

node -v

v12.14.1

which npm

/usr/local/bin/npm

npm -v

6.13.4

npm install --save core-js@^3

sudo npm install -g ember-cli

which ember

/usr/local/bin/ember

ember -v

ember-cli: 3.16.0

node: 12.16.1

os: darwin x64
